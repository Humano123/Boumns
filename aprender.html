@echo off
echo Olá!
set /p falar="-"

if /i "%falar%"=="ola" (
    echo Voce falou ola
) else if /i "%falar%"=="oi" (

) else (
    echo Voce falou outra coisa
)

If, Else if e Else em : 

#include <iostream>
#include  <string>

int main() {
    std::string valor;
    std::cout << "Olá!";

    std::cin >> valor;
    
    if (valor == "ola") {
        std::cout << "Você disse ola!";
    } else if (valor == "oi") {
        std::cout << "Você disse oi!";
    } else {
        std::cout << "Você disse outra coisa!";
    }
    
    return 0;
}

    printf("Olá!");
